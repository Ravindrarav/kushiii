#Grocery Shopping List
#Initialize the grocery list
grocery_list=["rice","pasta","bread","oats"]
def display_menu():
    print("\n---Grocery Shopping List Menu---")
    print("1. Display list")
    print("2. Add item")
    print("3. Insert item at a position")
    print("4. Remove item")
    print("5. Update/Replace item")
    print("6. Search for item")
    print("7. Sort list(A-Z)")
    print("8. Reverse List")
    print("9. Exit")
    print("==================================")
display_menu()
while True:
    choice=input("Enter your choice(1-9):")
    if choice=='1':
        if grocery_list:
            print("current Grocery List:",grocery_list)
        else:
            print("The grocery list is empty")
            
    elif choice=='2':
        item=input("Enter item to add:")
        grocery_list.append(item)
        print(f"'{item}'added to the list")
        print("Grocery list after adding:",grocery_list)
        print("------------------------")
        
    elif choice=='3':
        item=input("Enter item to insert:")
        pos=int(input("Enter position (starting from 0):"))
        if 0<=pos<=len(grocery_list):
            grocery_list.insert(pos,item)
            print(f"'{item}' inserted at position {pos}")
            print("Grocery list after inserting:",grocery_list)
            print("----------------------")
        else:
            print("Invalid position!")
    
    elif choice=='4':
        item=input("Enter item to remove:")
        if item in grocery_list:
            grocery_list.remove(item)
            print(f"'{item}'removed from the list")
            print("Grocery list after removing:",grocery_list)
        else:
            print(f"'{item}'not found in the list")
            
    elif choice=='5':
        old_item=input("Enter item to replace:")
        if old_item in grocery_list:
            new_item=input("Enter new item:")
            index=grocery_list.index(old_item)
            grocery_list[index]=new_item
            print(f"'{old_item}' replaces with'{new_item}'")
            print("Grocery list after replacing:",grocery_list)
        else:
            print(f"'{old_item}'not found in the list")
            
    elif choice=='6':
        item=input("Enter item to search:")
        if item in grocery_list:
            print(f"'{item}' found at index {grocery_list.index(item)}")
        else:
            print(f"'{item}' not found in the list")
            
    elif choice=='7':
        grocery_list.sort()
        print("list sorted in alphabetical order")
        print("Sorted Grocery List:",grocery_list)
        
    elif choice=='8':
        grocery_list.reverse()
        print("List order reversed")
        print("Reversed Grocery List:",grocery_list)
        
    elif choice=='9':
        print("Exiting program. Happy Shopping!")
        break
    
    else:
        print("Invalid choice! Please enter 1-9")
