books_available = ["python", "database management", "operating system", "computer networks", "open source tools"]
books_taken = []
books_returned = []
books_to_return = []

while True:
    print("\n=== Library Management ===")
    print("1. Display available books")
    print("2. Issue a book")
    print("3. Return a book")
    print("4. Display books to be returned")
    print("5. Exit")

    choice = input("Enter your choice: ")

    if choice == "1":
        print("\nAvailable books:")
        for book in books_available:
            print("-", book)

    elif choice == "2":
        print("\nIssue a book:")
        book_name = input("Enter the book to be issued: ")
        if book_name in books_available:
            books_available.remove(book_name)
            books_taken.append(book_name)
            print(f"{book_name} has been issued successfully.")
        else:
            print("Sorry, the book is not available.")

    elif choice == "3":
        print("\nReturn a book:")
        book_name = input("Enter the book you want to return: ")
        if book_name in books_taken:
            books_taken.remove(book_name)
            books_available.append(book_name)
            books_returned.append(book_name)
            print(f"{book_name} has been returned successfully.")
        else:
            print("You have not taken this book.")

    elif choice == "4":
        print("\nBooks to be returned:")
        books_to_return.clear() 
        for book in books_taken:
            if book not in books_returned:
                books_to_return.append(book)
                print("-", book)

        if not books_to_return:
            print("No books currently need to be returned.")

    elif choice == "5":
        print("Exiting Library Management System...")
        break

    else:
        print("Invalid choice, try again.")
