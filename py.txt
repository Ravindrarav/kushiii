subtotal = 0

n = int(input("Enter number of items: "))

for i in range(n):
    name = input("Enter item name: ")
    price = float(input("Enter item price: "))
    qty = int(input("Enter quantity: "))

    total = price * qty
    subtotal = subtotal + total

discount = subtotal * 0.10
after_discount = subtotal - discount

gst = after_discount * 0.05
final = after_discount + gst

print("Discount =", discount)
print("GST =", gst)

# Base class
class Item:
    def __init__(self, name, price):
        self.name = name
        self.price = price

    def apply_discount(self):
        return self.price   # No discount


# Subclass Electronics
class Electronics(Item):
    def apply_discount(self):
        return self.price * 0.90   # 10% discount


# Subclass Clothing
class Clothing(Item):
    def apply_discount(self):
        return self.price * 0.80   # 20% discount


# Main program
e = Electronics("Laptop", 1000)
c = Clothing("T-Shirt", 50)

print("Laptop discounted price:", e.apply_discount())
print("T-Shirt discounted price:", c.apply_discount())
